With JavaScript:

function toggleRemoveChild(id) {
    const cb  = document.getElementById('del-' + id);
    const row = document.getElementById('child-row-' + id);
    const btn = document.getElementById('btn-del-' + id);
    cb.checked = !cb.checked;
    if (cb.checked) {
        row.style.opacity = '0.4';
        btn.textContent   = 'Undo';
        btn.classList.replace('btn-outline-danger', 'btn-outline-secondary');
    } else {
        row.style.opacity = '1';
        btn.textContent   = 'Remove';
        btn.classList.replace('btn-outline-secondary', 'btn-outline-danger');
    }
}
d) Add client-side form validation on submit — highlight invalid fields and scroll to the first error:

document.getElementById('householdForm').addEventListener('submit', function(e) {
    let valid = true;
    this.querySelectorAll('.is-invalid').forEach(el => el.classList.remove('is-invalid'));
    ['first_name', 'last_name', 'phone'].forEach(id => {
        const el = document.getElementById(id);
        if (el && !el.value.trim()) { el.classList.add('is-invalid'); valid = false; }
    });
    const mp = document.getElementById('monthly_payment');
    if (mp && (isNaN(parseFloat(mp.value)) || parseFloat(mp.value) < 25)) {
        mp.classList.add('is-invalid'); valid = false;
    }
    document.querySelectorAll('#newChildrenContainer .child-entry').forEach(entry => {
        const n = entry.querySelector('input[name^="new_child_name_"]');
        if (n && !n.value.trim()) { n.classList.add('is-invalid'); valid = false; }
    });
    if (!valid) {
        e.preventDefault();
        const first = document.querySelector('.is-invalid');
        if (first) first.scrollIntoView({ behavior: 'smooth', block: 'center' });
    }
});
